const sortByLength = (array) => {
    let result = [];
    for (let i = 0; i < array.length; i++) {
        let minSizeItem = array.reduce((last, item) => {
            if(last.length < item.length) {
                return last;
            } else {
                last = item;
                return last;
            }
        });
    let index = array.indexOf(minSizeItem);
    result.push(minSizeItem);
    delete array[index];
    }
  return result;
}
